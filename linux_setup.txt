setup nfs booting:
in /export
sudo ln -s /home/jima/work/__Buildxxx/ 8280
or
sudo mount --bind /home/jima/work/__Build_8360__ 8360
(NOTE: bind is not persistent across boots)

in /tftpboot
sudo ln -s /home/jima/work/__Buildxxx/binaries 8280
sudo apt-get install nfs-server
sudo apt-get install xinetd tftpd tftp -y

add to /etc/exports:
/export/8360 *(rw,sync,no_subtree_check,no_root_squash)


----
set up tftp
sudo vi /etc/xinetd.d/tftp
service tftp
{
   protocol = udp
   port = 69
   socket_type = dgram
   wait = yes
   user = nobody
   server = /usr/sbin/in.tftpd
   server_args = tftpboot -s
   disable = no
}

Now letâ€™s change the ownership of the directory:

sudo mkdir /tftpboot
sudo chown -R nobody:nobody /tftpboot
# NOTE: on mint18.3 I only did nobody (without second nobody)
sudo chmod -R 777 /tftpboot

and start the TFTP service:

sudo service xinetd stop
sudo service xinetd start
