set up tftp
sudo vi /etc/xinetd.d/tftp
service tftp
{
   protocol = udp
   port = 69
   socket_type = dgram
   wait = yes
   user = nobody
   server = /usr/sbin/in.tftpd
   server_args = tftpboot -s
   disable = no
}
